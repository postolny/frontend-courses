  <!-- здесь this ссылается на саму кнопку, которую нажали -->
  <button onclick="toggle(this)">Секция 1</button>
  <div class="content">Содержимое секции 1</div>

  <button onclick="toggle(this)">Секция 2</button>
  <div class="content">Содержимое секции 2</div>

  <button onclick="toggle(this)">Секция 3</button>
  <div class="content">Содержимое секции 3</div>

</div>

<script>

// передаём в функцию toggle() нажатую кнопку из toggle(this)
function toggle(btn) {
  
  // получаем следующий сразу за кнопкой div
  const content = btn.nextElementSibling;

  // при повторном нажатии на кнопку закрываем открытый блок
  // если выбранный блок уже открыт (имеет класс open)
  if (content.classList.contains("open")) {
    // закрываем его (удаляем у него класс open)
    content.classList.remove("open");
    return; // выходим из функции
  }

  // закрываем все открытые блоки, обходя их циклом и удаляя класс open
  document.querySelectorAll(".content").forEach(item => {
    item.classList.remove("open");
  });

  // открываем только нажатый, добавляя класс open
  content.classList.add("open");
}
