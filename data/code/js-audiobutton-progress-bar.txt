<script>
const player = document.getElementById("player");
const playPause = document.getElementById("play-pause");

const progressBar = document.getElementById("progressBar");
const progressFill = document.getElementById("progressFill");

playPause.addEventListener("click", () => {
  if (player.paused) {
    player.play();
  } else {
    player.pause();
  }
});

player.addEventListener("play", () => {
  playPause.classList.add("playing");
});

player.addEventListener("pause", () => {
  console.log(player.currentTime);
  playPause.classList.remove("playing");
});

player.addEventListener("ended", () => {
  player.currentTime = 0;
  playPause.classList.remove("playing");
});

// в обработчике событие timeupdate срабатывает каждый раз, когда во время воспроизведения меняется текущее время
player.addEventListener("timeupdate", () => {
  // чтобы вычислить какая часть трека уже воспроизведена делим текущее время (currentTime) на общую продолжительность трека (duration) и получаем число от 0 до 1, результат умножаем на 100, чтобы перевести в проценты
  const percent = (player.currentTime / player.duration) * 100;
  // изменяем CSS-свойство width у .progress-fill, присваивая %
  progressFill.style.width = percent + "%";
  // или с шаблонной строкой
  // progressFill.style.width = `${percent}%`;
});
