<!doctype html>
<html>
<head>
<style>
body {
  background: white; /* необязательно */
  color: #555;
  /* плавный переход цвета фона и цвета текста */
  transition: background-color 0.3s, color 0.3s;
}

body.dark {
  background: #353535;
  color: white;
}
</style>
</head>
<body>

<button class="btn">Тёмная тема</button>

<p>Абзац</p>

<script>

let btn = document.querySelector(".btn"); // находим первый элемент с классом btn

// classList - это свойство DOM-элемента, которое позволяет управлять CSS-классами с помощью методов add(), remove(), toggle() и replace()
// метод toggle() переключает CSS-класс
btn.addEventListener("click", () => {
  document.body.classList.toggle("dark");
  
  // проверяем имеет ли body класс dark с помощью метода contains()
  // переключаем текст кнопки с помощью тернарного оператора
  btn.textContent = document.body.classList.contains("dark") ? "Светлая тема" : "Тёмная тема";
});

</script>

</body>
</html>
