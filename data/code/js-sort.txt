<ul id="result"></ul>

<script>
const products = [
  { name: "Молоко", price: 39.99 },
  { name: "Лук-репка в сетке", price: 46.99 },
  { name: "Морковь весовая", price: 38.99 },
  { name: "Крупа горох", price: 33.99 },
  { name: "Яблоко Сезонное фасованное", price: 99.99 },
];

const btn = document.getElementById("btn");
const btn1 = document.getElementById("btn1");
const result = document.getElementById("result");
const list = document.getElementById("list");
const btn2 = document.getElementById("btn2");

// оборачиваем в функцию
function render() {
  list.innerHTML = ""; // очищаем
  // метод forEach() перебирает каждый товар массива products
  // каждый товар по очереди попадает во временную переменную product
  products.forEach(function (product) {
    // добавляем каждый товар отдельным элементом списка <li> (раз мы выводим HTML, то используем свойство innerHTML)
    // += (оператор присваивания с добавлением) добавляет новый HTML к уже существующему содержимому, поэтому все товары будут выведены один за другим
    list.innerHTML += `<li>${product.name} - ${product.price} ₽</li>`;
  });
}

render(); // и запускаем её

/*
sort() изменяет исходный массив products
products.sort((a, b) => a.price - b.price);

чтобы не менять исходный массив, сначала нужно создать копию и сортировать уже её
const cheapToExpensive = [...products].sort(function (a, b) {
  return a.price - b.price;
});
или со стрелочной функцией
const cheapToExpensive = [...products].sort((a, b) => a.price - b.price);
*/

btn.addEventListener("click", function () {
  const cheapToExpensive = products.sort(function (a, b) {
    return a.price - b.price;
  });

  result.innerHTML = ""; // очищаем

  cheapToExpensive.forEach(function (product) {
    result.innerHTML += `<li>${product.name} - ${product.price} ₽</li>`;
  });
});

btn1.addEventListener("click", function () {
  const expensiveToCheap = products.sort(function (a, b) {
    return b.price - a.price;
  });

  result.innerHTML = "";

  expensiveToCheap.forEach(function (product) {
    result.innerHTML += `<li>${product.name} - ${product.price} ₽</li>`;
  });
});

btn2.addEventListener("click", function () {
  render(); // обновляем
});
</script>
