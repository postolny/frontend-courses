<script>

const result = document.querySelector("#result");
const input = document.querySelector("#number");

function renderTable() {
  result.innerHTML = ""; // очищаем таблицу

  const number = Number(input.value); // input.value всегда строка, поэтому мы преобразуем её в число с помощью Number(), чтобы можно было выполнять вычисления

  let i = 1; // создаём счётчик (начинаем с 1)

  while (i <= 10) {
    // создаём строку
    const tr = document.createElement("tr");

    // создаём ячейки
    const td1 = document.createElement("td");
    const td2 = document.createElement("td");
    const td3 = document.createElement("td");

    // заполняем текстом
    td1.textContent = number + " × " + i;
    // td1.textContent = `${number} × ${i}`; // с шаблонной строкой
    td2.textContent = "=";
    td3.textContent = number * i;

    // добавляем ячейки в строку
    tr.append(td1, td2, td3);

    // добавляем строку в таблицу
    result.append(tr);

    i++; // увеличиваем счётчик на единицу
  }
}

renderTable(); // запускаем

input.addEventListener("input", renderTable); // обновление при вводе
// input.addEventListener("change", renderTable); // сработает при нажатии enter или при клике вне поля
