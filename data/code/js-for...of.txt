// По нажатию кнопки обходим каждый блок .colored-box и меняем цвет фона на золотой
const btn = document.querySelector("#btn");
const coloredBoxes = document.querySelectorAll(".colored-box");

btn.addEventListener("click", () => {
  for (const coloredBox of coloredBoxes) {
    coloredBox.style.backgroundColor = "gold";
  }
});
