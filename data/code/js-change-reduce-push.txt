let cart = [];

function updateCart() {
  const cartList = document.getElementById("cartList");
  const cartCount = document.getElementById("cartCount");
  const cartTotal = document.getElementById("cartTotal");

  cartCount.textContent = cart.length;

  cartList.innerHTML = "";
  
  // для каждого элемента массива cart создаётся временная переменная sheetMusic и выполняется код внутри
  cart.forEach(sheetMusic => {
    cartList.innerHTML += `
      <li>
        ${sheetMusic.author} - ${sheetMusic.title}
        (${sheetMusic.price} ₽)
      </li>
    `;
  });
  // или через цикл for
  // for (let i = 0; i < cart.length; i++) {
  //   cartList.innerHTML += `
  //     <li>
  //       ${cart[i].author} - ${cart[i].title}
  //       (${cart[i].price} ₽)
  //     </li>
  //   `;
  // }
  
  // берём массив cart, проходимся по каждому элементу, складывем цены книг, начиная с нуля, и результат записываем в total (sheetMusic - это текщий элемент массива, sum - накопитель - он хранит результат предыдущего шага, 0 - это начальное значение sum)
  const total = cart.reduce((sum, sheetMusic) => {
    return sum + sheetMusic.price;
  }, 0);
  
  // или через for (переменная i хранит индекс)
  // let total = 0;
  // for (let i = 0; i < cart.length; i++) {
  //   total += cart[i].price;
  // }

  // или через for...of (здесь sheetMusic (хранит сам объект) - это тоже временная переменная как и в reduce)
  // let total = 0;
  // for (let sheetMusic of cart) {
  //   total += sheetMusic.price;
  // }

  // выводим результат из total
  cartTotal.textContent = total;
}

select.addEventListener("change", function () {
  /* this ссылается на элемент, на котором произошло событие, то есть на <select>, эквивалентно записи const value = select.value; или const value = document.getElementById("composer").value;

  Поскольку стрелочная функция не имеет this, то в ней можно использовать свойство target:
  select.addEventListener("change", (e) => {
    const value = e.target.value;
  }
  */
  const value = this.value;
  
  // Если <option value="">, то очищаем результат и выходим из функции
  if (value === "") {
    result.innerHTML = "";
    return;
  }
  
  // при выборе в селекте Чайковский в переменную value записывается tchaikovsky, затем с помощью composers[value] в объекте composers находится ключ tchaikovsky, а его значение записывается в переменную composer
  const composer = composers[value];

  // берём значение каждого свойства объекта composer (image, author, title и т.д.) и подставляем его в шаблонную строку
  result.innerHTML = `
    <div class="card">
      <img class="sheet-music-image" src="${composer.image}">
      <p>Композитор: ${composer.author}</p>
      <p>Название: ${composer.title}</p>

      <p>Издательство: ${composer.publisher}</p>
      <p>Год издания: ${composer.year}</p>
      <p>Страниц: ${composer.pages}</p>
      <p>ISMN: ${composer.ismn}</p>
      <p>Формат: ${composer.format}</p>

  `;

    // запускаем функцию updateCart()
    updateCart();
  });
});
