<script>
const btn = document.getElementById("calcBtn");
const result = document.getElementById("result");

const prices = [39.99, 46.99, 38.99];

btn.addEventListener("click", function () {
  let sum = 0;

  for (let i = 0; i < prices.length; i++) {
    // обращаемся к элементу массива по индексу и с каждым проходом прибавляем текущее значение sum к текущему элементу массива
    sum += prices[i];
    // эквивалентно sum = sum + prices[i];
  }

  result.textContent = "Итого к оплате: " + sum + " рублей";
  // или в обратных кавычках (шаблонная строка)
  // result.textContent = `Итого к оплате: ${sum} рублей`;
});
</script>
