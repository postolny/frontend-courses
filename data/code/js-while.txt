<script>
const btn = document.querySelector("#btn");
const result = document.querySelector("#result");

let isVisible = false; // флаг (изначально блок скрыт, поэтому значение false)

let i = 1; // создаём счётчик (начинаем с 1)
result.innerHTML = ""; // чтобы старый результат не оставался, очищаем его при клике
  
// цикл будет повторяться, пока i меньше или равно 10
while (i <= 10) {
  result.innerHTML += "5 × " + i + " = " + (5 * i) + "<br>"; // добавляем перенос, чтобы было в столбик
  // с шаблонной строкой
  // result.innerHTML += `5 × ${i} = ${5 * i}<br>`;
  i++; // увеличиваем счётчик на единицу
}

btn.addEventListener("click", () => {
  isVisible = !isVisible; // переключатель (true - false)
  
  // если isVisible равно true, выбираем block, иначе none
  result.style.display = isVisible ? "block" : "none";
  // если isVisible равно true, выбираем Скрыть таблицу, иначе Показать таблицу
  btn.textContent = isVisible ? "Скрыть таблицу" : "Показать таблицу";
});
