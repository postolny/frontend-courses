<!doctype html>
<html>
<body>

<ul id="list"></ul>

<button id="btn">Показать товары дешевле 40 ₽</button>

<ul id="result"></ul>

<script>
const products = [
  {name: "Молоко", price: 39.99},
  {name: "Лук-репка в сетке", price: 46.99},
  {name: "Морковь весовая", price: 38.99},
  {name: "Крупа горох", price: 33.99},
  {name: "Яблоко Сезонное фасованное", price: 99.99}
];

const btn = document.getElementById("btn");
const result = document.getElementById("result");
const list = document.getElementById("list");

// метод forEach() перебирает каждый товар массива products
// каждый товар по очереди попадает во временную переменную product
products.forEach(function (product) {
// добавляем каждый товар отдельным элементом списка <li> (раз мы выводим HTML, то используем свойство innerHTML)
// составной оператор присваивания += (оператор присваивания с добавлением) добавляет новый HTML к уже существующему содержимому, поэтому все товары будут выведены один за другим
  list.innerHTML += `
    <li>${product.name} - ${product.price} ₽</li>
  `;
});

btn.addEventListener("click", function () {
  // фильтруем массив
  // метод filter() проходит по каждому элементу массива products; каждый элемент по очереди передаётся в параметр (временную переменную) product; если функция возвращает true, товар попадёт в новый массив filteredProducts; если false - товар будет пропущен
  const filteredProducts = products.filter(function (product) {
    return product.price < 40;
  });
  // или
  //const filteredProducts = products.filter(product => product.price < 40);

  // очищаем список result перед выводом
  result.innerHTML = "";
  filteredProducts.forEach(function (product) {
    result.innerHTML += `
      <li>${product.name} - ${product.price} ₽</li>
    `;
  });
});
</script>

</body>
</html>
