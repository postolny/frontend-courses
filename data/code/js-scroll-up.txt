<script>

const btn = document.getElementById("scrollUpBtn");

window.addEventListener("scroll", () => {
  // если прокрутили больше, чем на 300px
  if (window.scrollY > 300) {
    // раньше использовали document.documentElement.scrollTop
    // или document.body.scrollTop
    console.log(window.scrollY);
    // добавляем кнопке класс show
    btn.classList.add("show");
  } else {
    // иначе удаляем класс show
    btn.classList.remove("show");
  }
});

btn.addEventListener("click", () => {
  window.scrollTo({
    top: 0,
    behavior: "smooth"
  });
});

// создаём 30 абзацев
const content = document.getElementById("content");

for (let i = 1; i <= 30; i++) {
  const p = document.createElement("p");
  p.textContent = "Абзац " + i +". Пусть всегда будет солнце, пусть всегда будет небо, пусть всегда будет мама, пусть всегда буду я.";
  content.appendChild(p);
}
