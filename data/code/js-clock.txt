<div id="clock"></div>

<script>

const clock = document.getElementById("clock");

function showTime() {

  const now = new Date();
  
  // получаем часы, минуты и секунды из объекта Date
  let hours = now.getHours();
  let minutes = now.getMinutes();
  let seconds = now.getSeconds();

  // если число меньше 10, добавляем впереди 0
  if (hours < 10) {
      hours = "0" + hours;
  }

  if (minutes < 10) {
      minutes = "0" + minutes;
  }

  if (seconds < 10) {
      seconds = "0" + seconds;
  }

  clock.innerHTML = "<p>" + hours + ":" + minutes + ":" + seconds + "</p>";
  // или с использованием шаблонной строки
  // clock.innerHTML = `<p>${hours}:${minutes}:${seconds}</p>`;
  clock.style.color = "#16a085";
  clock.style.fontSize = "40px";
  clock.style.fontWeight = "bold";
  clock.style.textAlign = "center";
}

showTime(); // вызываем функцию после загрузки страницы

setInterval(showTime, 1000); // обновляем каждую секунду

</script>

</body>
