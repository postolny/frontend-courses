  <button class="btn"><span>Секция 3</span><span class="chevron">▼</span></button>

<script>

// проходим по каждому элементу с классом .btn и при клике вызываем функцию toggle
document.querySelectorAll(".btn").forEach(button => {
  button.addEventListener("click", toggle);
});

function toggle(event) {
  // event (назвать можно как угодно) - информация о произошедшем клике, переданная в функцию
  // получаем кнопку, на которой сработал обработчик события, используя свойство currentTarget
  // если в кнопке нет вложенных элементов, то target и currentTarget будут указывать на один и тот же элемент, поэтому можно использовать target, но если внутри кнопки есть вложенный элемент, например <span>, то при клике на него target будет ссылаться на этот вложенный элемент, и nextElementSibling будет искать следующий элемент уже у него, а не у кнопки
  const btn = event.currentTarget;
  // получаем следующий сразу за кнопкой div
  const content = btn.nextElementSibling;
  // сохраненяем результат проверки в переменную isOpen - она получает true, если btn имеет класс active
  const isOpen = btn.classList.contains("active");

  // удаляем у всех кнопок класс active
  document.querySelectorAll(".btn").forEach(item => {
      item.classList.remove("active");
  });

  // при повторном нажатии на кнопку закрываем открытый блок
  // если выбранный блок уже открыт (имеет класс open)
  if (content.classList.contains("open")) {
    // закрываем его (удаляем у него класс open)
    content.classList.remove("open");
    return; // выходим из функции
  }

  // закрываем все открытые блоки, обходя их циклом и удаляя класс open
  document.querySelectorAll(".content").forEach(item => {
    item.classList.remove("open");
  });

  // оператор логического отрицания !
  // если кнопка была закрыта, то открываем её (логика на основе булевого значения)
  if (!isOpen) {
    btn.classList.add("active");
    // открываем только нажатый, добавляя класс open
    content.classList.add("open");
  }
}

</script>
